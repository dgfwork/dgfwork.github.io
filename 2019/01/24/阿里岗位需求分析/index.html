阿里岗位需求分析

Posted on 2019-01-24 | In spider

本想写个通用爬虫，然后自行根据岗位需求抓取但是不知道url的keyword是如何编码的，链接如下：

https://job.alibaba.com/zhaopin/positionList.htm?keyWord=cHl0aG9u&_input_charset=UTF-8&

另外阿里的反爬措施做得还是比较到位的，直接跳转出现乱码会被屏蔽
爬虫代码如下：

#coding:utf-8
#解决编码问题
import sys
reload(sys)
sys.setdefaultencoding('utf8')
import  selenium
from selenium import webdriver
import selenium.webdriver.common.keys
import time
import lxml
import lxml.etree
import urllib2

"""
抓取阿里的岗位python需求主要是解决网页点击链接不跳转的情况
其他的情况还是先提出每一页的若干岗位需求链接
再逐个访问提取岗位需求
第一页是入口,进去之后抓到总的页面数
再用selenium模拟点击相应的次数抓取每页里面的链接
再访问逐个链接抓取岗位需求制作词云
"""
def geturllist():
    driver= webdriver.Chrome()
    driver.get("https://job.alibaba.com/zhaopin/positionList.htm?keyWord=cHl0aG9u&_input_charset=UTF-8")
    time.sleep(5)
    html = driver.page_source
    html = lxml.etree.HTML(html)
    pagenumbers = eval(html.xpath("//*[@id=\"J-            pagination\"]//ul/li[last()-1]/@data-index")[0])
    urllist = []
    url = html.xpath("//*[@id=\"J-list-box\"]//tr//td/span/a/@href")
    for data in url:
        urllist.append("https://job.alibaba.com/zhaopin/"+data)
    #访问第一页之后后面每页要模拟手工点击
    for i in range(2,pagenumbers+1):
        elem=driver.find_element_by_xpath("//*[@class=\"pagination\"]//ul//li[last()]//a")
        elem.click()  #模拟点击
        time.sleep(3)
        html = driver.page_source
        html = lxml.etree.HTML(html)
        tempurl = html.xpath("//*[@id=\"J-list-box\"]//tr//td/span/a/@href")
        for data in tempurl:
            urllist.append("https://job.alibaba.com/zhaopin/" + data)
    return urllist
    driver.close()
def getinfo(url):
    headers = {"User-Agent": "Mozilla/5.0 (compatible; MSIE 9.0; Windows NT 6.1; Trident/5.0);"}
    request = urllib2.Request(url, headers=headers)  # 发起请求，
    # 也可以通过调⽤Request.add_header() 添加/修改⼀个特定的 header
    request.add_header("Connection", "keep-alive")  # 一直活着
    response = urllib2.urlopen(request)
    data = response.read()  # 打开请求，抓取数据
    html = lxml.etree.HTML(data)
    #xpath提取信息
    info = html.xpath("//*[@class=\"detail-content\"]/text()")
    return info


filepath = "info.txt"
file = open(filepath,"w")
for url in geturllist():
    for line in getinfo(url):
        file.write(line.strip()+"\r\n")
file.close()

以上抓取到文本写入info.txt

#coding:utf-8
import jieba  #分词
import matplotlib
import matplotlib.pyplot as plt
import wordcloud
from wordcloud import WordCloud,ImageColorGenerator,STOPWORDS#词云
import numpy as np #科学计算
import jieba.analyse
from PIL import Image  #处理图片
textfile=open("info.txt").read()
textfile = textfile.replace("1","").replace("2","").replace("3","").replace("4","")
textfile = textfile.replace("5","").replace("6","").replace("7","").replace("8","")
wordlist=jieba.cut_for_search(textfile)
space_list=" ".join(wordlist)
backgroud=np.array(Image.open("2.jpg")) #背景图片

                      mask=backgroud,#写字用的背景图，从背景图取颜色

                      stopwords=STOPWORDS, #停止的默认词语
                      font_path="simkai.ttf", #字体
                      max_font_size=200, #最大字体尺寸
                      random_state=100,#随机角度
                      scale=2).generate(space_list) #生成词云

image_color=ImageColorGenerator(backgroud) #生成词云的颜色
plt.imshow(mywordcloud) #显示词云
plt.axis("off") #关闭保存
plt.show()
